### `<Waypoint>`

`Waypoint` does something crazy.

#### Render Props

```tsx
<Scrollytelling.Waypoint
  at: number; // Scrolled position at which to trigger the waypoint
  children?: AnimationProps["children"]; // Optional: children to animate
  tween?: SimpleTween & { target?: TweenTarget }; // Optional: tween config and target, to apply when the waypoint is triggered
  onCall?: (() => void) | undefined; // Optional: called when the waypoint is triggered
  onReverseCall?: (() => void) | undefined; // Optional: called when the waypoint is triggered in reverse
  label?: string | undefined; // Optional: label for the waypoint
  disabled?: boolean | undefined; // Optional: whether the waypoint is disabled
/>
```

#### Usage

```jsx copy
import * as Scrollytelling from "@bsmnt/scrollytelling";

export const HomePage = () => (
  <Scrollytelling.Root>
    <div id="fun-div">
      <Scrollytelling.Waypoint at={50} onCall={triggerConfetti} />
      <Scrollytelling.Waypoint
        at={75}
        tween={{
          target: ["#fun-div"],
          to: { background: "black", color: "white" },
          duration: 0.35,
        }}
      />
    </div>
  </Scrollytelling.Root>
);
```
